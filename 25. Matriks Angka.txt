import java.util.*;
class Main {
  public static void main(String[] args) {
    
    Scanner input = new Scanner(System.in);
    
    int c = input.nextInt();
    for (int a = 0; a < c; a++) {
      for (int b = 0; b < c; b++) {
        System.out.print(b + " ");
      }
      System.out.println();
      }
    }
  }
